#include <conio.h>
#include <windows.h>

}

main() {
	
    int J, K, T, F, N ;
    char OP = 'S';

          {
           system("cls");

           do {
              gotoxy(25,8); cout << "Digite el Numero de Calculo : ";

              gotoxy(25,22); cout <<"ERROR.....EL VALOR DEBE SER ENTRE 1 y 12";
           }  while(( N<= 0 )||( N>12 ));

           gotoxy(31,5); cout << "TABLA DE MULTIPLICAR DEL " <<  N;
